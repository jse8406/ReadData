# 나라장터 크롤러 + DB 구현 계획

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 나라장터(g2b.go.kr)에서 개찰결과를 크롤링하여 기초금액, 예정가격, 예비가격 15건, 개찰일시를 수집하고 SQLite DB에 저장한다.

**Architecture:** Selenium으로 나라장터 접속 → WebSquare 메뉴 이동 → 목록 API(selectOnbsRsltClsfList.do)로 개찰완료 건 수집 → 예비가격 API(selectRsvePrceInfo.do)로 상세 데이터 추출 → 물품/용역 구분하여 단일 DB에 저장. 1년치 데이터를 날짜 구간(1개월 단위)으로 나눠서 페이지네이션 처리.

**Tech Stack:** Python 3, Selenium, chromedriver-autoinstaller, SQLite3

```python
# db/g2b_module.py
import os
import sqlite3


class G2bDB:
    def __init__(self, db_name):
        self.db_name = db_name

    def init_db(self):
        if os.path.isfile(self.db_name):
            print("DB : g2b Database exists!")
            return
        conn = sqlite3.connect(self.db_name)
        cursor = conn.cursor()
        cursor.execute("""
            CREATE TABLE bid_results (
                bid_pbanc_no TEXT NOT NULL,
                bid_pbanc_ord TEXT NOT NULL,
                bid_clsf_no TEXT NOT NULL,
                prcm_bsne_se_cd TEXT NOT NULL,
                bid_pbanc_nm TEXT,
                date TEXT NOT NULL,
                base_amount INTEGER NOT NULL,
                predict_price INTEGER NOT NULL,
                rsve_price_1 INTEGER,
                rsve_price_2 INTEGER,
                rsve_price_3 INTEGER,
                rsve_price_4 INTEGER,
                rsve_price_5 INTEGER,
                rsve_price_6 INTEGER,
                rsve_price_7 INTEGER,
                rsve_price_8 INTEGER,
                rsve_price_9 INTEGER,
                rsve_price_10 INTEGER,
                rsve_price_11 INTEGER,
                rsve_price_12 INTEGER,
                rsve_price_13 INTEGER,
                rsve_price_14 INTEGER,
                rsve_price_15 INTEGER,
                PRIMARY KEY (bid_pbanc_no, bid_pbanc_ord, bid_clsf_no)
            )
        """)
        conn.commit()
        cursor.close()
        conn.close()

    def insert_result(self, data: dict):
        conn = sqlite3.connect(self.db_name)
        cursor = conn.cursor()
        cursor.execute("""
            INSERT INTO bid_results (
                bid_pbanc_no, bid_pbanc_ord, bid_clsf_no, prcm_bsne_se_cd,
                bid_pbanc_nm, date, base_amount, predict_price,
                rsve_price_1, rsve_price_2, rsve_price_3, rsve_price_4,
                rsve_price_5, rsve_price_6, rsve_price_7, rsve_price_8,
                rsve_price_9, rsve_price_10, rsve_price_11, rsve_price_12,
                rsve_price_13, rsve_price_14, rsve_price_15
            ) VALUES (
                :bid_pbanc_no, :bid_pbanc_ord, :bid_clsf_no, :prcm_bsne_se_cd,
                :bid_pbanc_nm, :date, :base_amount, :predict_price,
                :rsve_price_1, :rsve_price_2, :rsve_price_3, :rsve_price_4,
                :rsve_price_5, :rsve_price_6, :rsve_price_7, :rsve_price_8,
                :rsve_price_9, :rsve_price_10, :rsve_price_11, :rsve_price_12,
                :rsve_price_13, :rsve_price_14, :rsve_price_15
            )
        """, data)
        conn.commit()
        cursor.close()
        conn.close()

    def exists(self, bid_pbanc_no, bid_pbanc_ord, bid_clsf_no):
        conn = sqlite3.connect(self.db_name)
        cursor = conn.cursor()
        cursor.execute(
            "SELECT 1 FROM bid_results WHERE bid_pbanc_no=? AND bid_pbanc_ord=? AND bid_clsf_no=?",
            (bid_pbanc_no, bid_pbanc_ord, bid_clsf_no)
        )
        result = cursor.fetchone()
        cursor.close()
        conn.close()
        return result is not None
```

### Task 2: 크롤러 구현 (`g2bBring.py`)

**Files:**
- Create: `g2bBring.py`

**핵심 로직:**
1. Selenium으로 나라장터 접속 → 개찰결과분류조회 메뉴 이동
2. 1개월 단위로 날짜 범위를 잡아 목록 API 호출 (페이지네이션 포함)
3. 각 개찰완료 건에 대해 예비가격 API 호출
4. 기초금액(bsamt), 예정가격(pnpr), 예비가격 15건이 모두 존재하는 경우만 DB 저장

**API 정리:**
- 목록: `POST /pb/pbo/pboc/OnbsRslt/selectOnbsRsltClsfList.do`
  - 요청: `bidPbancNo`, `onbsYmdStr`, `onbsYmdEnd`, `prcmBsneSeCd`, `lastIndex`, `firstIndex`, `pbancKndCd`
  - 응답: `result.onbsRsltClsfList[]` (bidPbancNo, bidPbancOrd, bidClsfNo, bidPrgrsOrd, prcmBsneSeCd, bidPgstCd, bidPbancNm, onbsPrnmntDt)

- 예비가격: `POST /pb/pbo/pboc/OnbsRslt/selectRsvePrceInfo.do`
  - 요청: `bidPbancNo`, `bidPbancOrd`, `bidClsfNo`, `bidPrgrsOrd`, `prcmBsneSeCd`
  - 응답: `bsisAmtInfo.bsamt`(기초금액), `bsisAmtInfo.pnpr`(예정가격), `rsvePrceList[]`(예비가격 15건), `pbancInfo.realOnbsPrnmntDt`(개찰일시)

- [ ] **Step 1: g2bBring.py 작성**

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""나라장터 개찰결과 크롤러 - 기초금액, 예정가격, 예비가격 15건 수집"""

import time
import json
import os
import sqlite3
from datetime import datetime, timedelta
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
import chromedriver_autoinstaller
from db.g2b_module import G2bDB

chromedriver_autoinstaller.install()
BASE_DIR = os.path.dirname(os.path.abspath(__file__))

# API URLs (브라우저 세션 내 JS fetch로 호출)
LIST_API = "/pb/pbo/pboc/OnbsRslt/selectOnbsRsltClsfList.do"
RSVE_API = "/pb/pbo/pboc/OnbsRslt/selectRsvePrceInfo.do"


def create_driver():
    options = webdriver.ChromeOptions()
    options.add_argument("headless")
    options.add_argument("--no-sandbox")
    options.add_argument("--disable-dev-shm-usage")
    options.add_argument("--window-size=1920,1080")
    driver = webdriver.Chrome(options=options)
    return driver


def navigate_to_search(driver):
    """나라장터 접속 → 개찰결과분류조회 페이지 이동"""
    driver.get("https://www.g2b.go.kr")
    time.sleep(6)

    for m in driver.find_elements(By.CSS_SELECTOR, "a.depth1"):
        if '입찰' in m.text:
            m.click(); time.sleep(1); break
    for m in driver.find_elements(By.CSS_SELECTOR, "a.depth2"):
        if '개찰' in m.text:
            m.click(); time.sleep(1); break
    for m in driver.find_elements(By.CSS_SELECTOR, "a.depth3"):
        if '개찰결과분류조회' in m.text:
            m.click(); time.sleep(8); break

    # WebSquare 로딩 대기
    for _ in range(15):
        try:
            ready = driver.execute_script(
                "return typeof WebSquare !== 'undefined' && WebSquare.util !== undefined;"
            )
            if ready:
                return True
        except:
            pass
        time.sleep(1)
    return False


def fetch_list(driver, date_start, date_end, first_index=1, last_index=100):
    """목록 API 호출 - JS XMLHttpRequest 사용"""
    payload = {
        "dlSrchOnbsRsltClsfListInM": {
            "bidPbancNo": "",
            "bidPbancOrd": "",
            "bidPbancNm": "",
            "onbsYmdStr": date_start,
            "onbsYmdEnd": date_end,
            "prcmBsneSeCd": "0000 \t070001 \t070002 \t070003 \t070004 \t070005",
            "frcpYn": "Y",
            "rsrvYn": "Y",
            "laseYn": "Y",
            "chkBsneAllYn": "Y",
            "lastIndex": last_index,
            "firstIndex": first_index,
            "inqEnd": "",
            "stepBarYn": "",
            "pbancKndCd": "\t440002"
        }
    }
    result = driver.execute_script("""
        var xhr = new XMLHttpRequest();
        xhr.open('POST', arguments[0], false);
        xhr.setRequestHeader('Content-Type', 'application/json;charset=UTF-8');
        xhr.send(arguments[1]);
        return xhr.responseText;
    """, LIST_API, json.dumps(payload, ensure_ascii=False))

    data = json.loads(result)
    return data.get("result", {}).get("onbsRsltClsfList", [])


def fetch_rsve_price(driver, item):
    """예비가격 상세 API 호출"""
    payload = {
        "dlSrchPbancInfoInM": {
            "bidPbancNo": item["bidPbancNo"],
            "bidPbancOrd": item["bidPbancOrd"],
            "bidClsfNo": item["bidClsfNo"],
            "bidPrgrsOrd": item["bidPrgrsOrd"],
            "prcmBsneSeCd": item["prcmBsneSeCd"],
            "blffVrfcYn": ""
        }
    }
    result = driver.execute_script("""
        var xhr = new XMLHttpRequest();
        xhr.open('POST', arguments[0], false);
        xhr.setRequestHeader('Content-Type', 'application/json;charset=UTF-8');
        xhr.send(arguments[1]);
        return xhr.responseText;
    """, RSVE_API, json.dumps(payload, ensure_ascii=False))

    return json.loads(result)


def parse_rsve_prices(rsve_list):
    """rsvePrceList에서 15개 가격 추출. 15개 미만이면 None 반환."""
    prices = []
    for row in rsve_list:
        if "plrlRsvePrce1" in row:
            prices.append(row["plrlRsvePrce1"])
        if "plrlRsvePrce2" in row:
            prices.append(row["plrlRsvePrce2"])
    if len(prices) != 15:
        return None
    return prices


def generate_month_ranges(start_date, end_date):
    """시작~종료 날짜를 1개월 단위 구간으로 분할"""
    ranges = []
    current = start_date
    while current < end_date:
        month_end = current.replace(day=28) + timedelta(days=4)
        month_end = month_end.replace(day=1) - timedelta(days=1)
        if month_end > end_date:
            month_end = end_date
        ranges.append((current.strftime("%Y%m%d"), month_end.strftime("%Y%m%d")))
        current = month_end + timedelta(days=1)
    return ranges


def main():
    # DB 초기화
    db_path = os.path.join(BASE_DIR, 'db', 'g2b.db')
    db = G2bDB(db_path)
    db.init_db()

    # 날짜 범위: 1년 전 ~ 오늘
    end_date = datetime.now()
    start_date = end_date - timedelta(days=365)
    month_ranges = generate_month_ranges(start_date, end_date)

    print(f"수집 기간: {start_date.strftime('%Y-%m-%d')} ~ {end_date.strftime('%Y-%m-%d')}")
    print(f"총 {len(month_ranges)}개 월별 구간")

    # 브라우저 시작
    driver = create_driver()
    try:
        print("나라장터 접속 중...")
        if not navigate_to_search(driver):
            print("페이지 로딩 실패")
            return

        total_saved = 0
        total_skipped = 0

        for idx, (d_start, d_end) in enumerate(month_ranges):
            print(f"\n[{idx+1}/{len(month_ranges)}] {d_start} ~ {d_end}")

            # 페이지네이션: 100건씩
            first_index = 1
            page_size = 100

            while True:
                items = fetch_list(driver, d_start, d_end,
                                   first_index=first_index, last_index=first_index + page_size - 1)

                if not items:
                    break

                print(f"  페이지 {first_index}~ : {len(items)}건")

                for item in items:
                    bid_no = item["bidPbancNo"]
                    bid_ord = item["bidPbancOrd"]
                    bid_clsf = item["bidClsfNo"]
                    pgst = item["bidPgstCd"]

                    # 개찰완료만
                    if "완료" not in pgst and pgst != "입160004":
                        continue

                    # 중복 체크
                    if db.exists(bid_no, bid_ord, bid_clsf):
                        total_skipped += 1
                        continue

                    # 예비가격 API 호출
                    try:
                        detail = fetch_rsve_price(driver, item)
                    except Exception as e:
                        print(f"    {bid_no}: 예비가격 조회 실패 - {e}")
                        continue

                    bsis = detail.get("bsisAmtInfo", {})
                    bsamt = bsis.get("bsamt")
                    pnpr = bsis.get("pnpr")
                    rsve_list = detail.get("rsvePrceList", [])
                    pbanc_info = detail.get("pbancInfo", {})
                    real_dt = pbanc_info.get("realOnbsPrnmntDt", "")

                    # 기초금액, 예정가격 존재 확인
                    if not bsamt or not pnpr:
                        continue

                    # 예비가격 15건 파싱
                    prices = parse_rsve_prices(rsve_list)
                    if prices is None:
                        continue

                    # 날짜 (년/월/일만)
                    date_str = real_dt.split(" ")[0] if real_dt else ""
                    if not date_str:
                        date_str = item.get("onbsPrnmntDt", "").split(" ")[0]

                    bid_name = pbanc_info.get("bidPbancNm", item.get("bidPbancNm", ""))

                    row = {
                        "bid_pbanc_no": bid_no,
                        "bid_pbanc_ord": bid_ord,
                        "bid_clsf_no": bid_clsf,
                        "prcm_bsne_se_cd": item["prcmBsneSeCd"],
                        "bid_pbanc_nm": bid_name,
                        "date": date_str,
                        "base_amount": bsamt,
                        "predict_price": pnpr,
                    }
                    for i, p in enumerate(prices, 1):
                        row[f"rsve_price_{i}"] = p

                    try:
                        db.insert_result(row)
                        total_saved += 1
                        print(f"    ✅ {bid_no} | {date_str} | 기초:{bsamt:,} | 예정:{pnpr:,} | {item['prcmBsneSeCd']}")
                    except sqlite3.IntegrityError:
                        total_skipped += 1

                    time.sleep(0.3)  # API 부하 방지

                if len(items) < page_size:
                    break
                first_index += page_size

        print(f"\n{'='*50}")
        print(f"완료! 저장: {total_saved}건, 스킵: {total_skipped}건")

    finally:
        driver.quit()


if __name__ == "__main__":
    main()
```
